PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX ex: <http://www.w3.org/2002/07/owl#>  # Sesuaikan dengan XML namespace Anda

SELECT ?suspect ?platform ?msgText
WHERE {
  ?message rdf:type ex:TextMessage ;
           ex:hasMessageText ?msgText ;
           ex:sendsTextMessage ?suspect ;
           ex:belongsToPlatform ?platform .

  FILTER (REGEX(?msgText, "end it", "i") ||
          REGEX(?msgText, "ruin your life", "i") ||
          REGEX(?msgText, "watch your back", "i") ||
          REGEX(?msgText, "just disappear", "i") ||
          REGEX(?msgText, "leak private", "i") ||
          REGEX(?msgText, "I will scam you", "i") || 
          REGEX(?msgText, "you fell for it", "i") || 
          REGEX(?msgText, "stupid buyer", "i") ||
          REGEX(?msgText, "say goodbye to your money", "i"))
}
ORDER BY ?suspect ?platform
