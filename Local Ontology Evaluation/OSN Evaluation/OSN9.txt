PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX ex: <http://www.w3.org/2002/07/owl#>

SELECT ?story ?content ?media ?timestamp ?location ?device ?ip ?music ?sticker ?postedBy
WHERE {
  ?story rdf:type ex:Stories .
  OPTIONAL { ?story ex:hasStoryContent ?content. }
  OPTIONAL { ?story ex:hasStoryMedia ?media. }
  OPTIONAL { ?story ex:hasStoryTimestamp ?timestamp. }
  OPTIONAL { ?story ex:hasMessageLocation ?location. }
  OPTIONAL { ?story ex:hasDeviceInfo ?device. }
  OPTIONAL { ?story ex:hasUserInteractionIP ?ip. }
  OPTIONAL { ?story ex:hasStoryMusic ?music. }
  OPTIONAL { ?story ex:hasStoryStickers ?sticker. }
  OPTIONAL { ?story ex:postedBy ?postedBy. }
}
