PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX ex: <http://www.w3.org/2002/07/owl#>  # Sesuaikan dengan XML namespace Anda

SELECT ?liveEvent ?liveStartTime ?liveEndTime 
       ?commentText ?commentTimestamp ?commentUser
WHERE {
    ?liveEvent rdf:type ex:Live ;
               ex:hasLiveStartTime ?liveStartTime ;
               ex:hasLiveEndTime ?liveEndTime .

    ?comment rdf:type ex:Comment ;
             ex:commentsOnContent ?liveEvent ;
             ex:hasCommentText ?commentText ;
             ex:hasCommentTimestamp ?commentTimestamp ;
             ex:commentedBy ?commentUser .
}
ORDER BY ?liveEvent ?commentTimestamp
