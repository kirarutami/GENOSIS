# Prefixes
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX ex: <http://www.w3.org/2002/07/owl#> # Sesuaikan jika base IRI berbeda

SELECT ?originalPost ?retweetAction ?retweeter ?retweetTimestamp ?retweetComment
WHERE {
  # Tentukan postingan asli yang ingin dilacak penyebarannya
  # Misalnya, kita lacak penyebaran post ex:Post_Amina001 (sebagai contoh)
  VALUES ?originalPost { ex:Post_Amina001 }

  # Cari semua aksi retweet/reblog yang menargetkan postingan asli
  ?retweetAction rdf:type ex:ReblogRetweetPost ;
                 ex:retweetsOriginalPost ?originalPost ; # Property baru
                 ex:retweetedBy ?retweeter ;           # Property baru
                 ex:hasRetweetTimestamp ?retweetTimestamp . # Property baru

  # Ambil komentar retweet (jika ada)
  OPTIONAL { ?retweetAction ex:hasRetweetComment ?retweetComment . } # Property baru
}
ORDER BY ?retweetTimestamp
