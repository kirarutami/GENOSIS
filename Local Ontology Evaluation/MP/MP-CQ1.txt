PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX ex: <http://www.w3.org/2002/07/owl#>

SELECT DISTINCT ?actionType ?actor ?target ?reason ?time ?platform
WHERE {
  {
    # Data laporan
    ?report rdf:type ex:Report ;
            ex:reportInitiator ?actor ;
            ex:reportedUser ?target ;
            ex:hasReportReason ?reason ;
            ex:hasInteractionTime ?time ;
            ex:belongsToPlatform ?platform .
    BIND("Report" AS ?actionType)
  }
  UNION
  {
    # Data pemblokiran
    ?block rdf:type ex:Block ;
           ex:blockInitiator ?actor ;
           ex:blockedUser ?target ;
           ex:hasBlockReason ?reason ;
           ex:hasInteractionTime ?time ;
           ex:belongsToPlatform ?platform .
    BIND("Block" AS ?actionType)
  }
}
ORDER BY ?time
